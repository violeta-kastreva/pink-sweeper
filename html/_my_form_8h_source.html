593 void MyForm::GenerateMatrixOfButtons(int rows, int cols, int numBombs, array <array <bool>^ >^ mineData, array <array <bool>^ >^ revealedData, array <array <bool>^ >^ flaggedData)
594 {
595 const int buttonSize = 28;
596 const int padding = 1;
597 const int leftTotalPadding = (formWidth - (cols * (buttonSize + padding) - padding)) / 2 - rows;
598 const int topTotalPadding = (formHeight - (rows * (buttonSize + padding) - padding)) / 2 - 20;
599 mineField = gcnew MineField(rows, cols, numBombs);
600 mineField->InitializeField();
601 gameTimer->Stop();
602 elapsedTimeInSeconds = 0;
603 for (int i = 0; i < rows; ++i) {
604 for (int j = 0; j < cols; ++j) {
605 MineButton^ button = gcnew MineButton(i, j);
606 button->Size = System::Drawing::Size(buttonSize, buttonSize);
607 button->Font = gcnew System::Drawing::Font(L"a_Rewinder", 15, System::Drawing::FontStyle::Regular);
608 button->ForeColor = System::Drawing::Color::White;
609 button->Location = Point(leftTotalPadding + padding + j * (buttonSize + padding), topTotalPadding + padding + i * (buttonSize + padding));
610 button->MouseDown += gcnew MouseEventHandler(this, &MyForm::MineButtonClick);
611 this->Controls->Add(button);
612 mineField->SetButton(i, j, button);
613
614 if (mineData != nullptr && mineData[i][j]) {
615 button->IsMine = true;
616 mineField->AddMine(i, j);
617 button->AdjacentMines = mineField->GetAdjacentMineCount(i, j);
